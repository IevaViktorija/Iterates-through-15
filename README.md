# Iterates-through-15
```java
public class Main {
    public static void main(String[] args) {
        for (int i = 0; i <= 15; i++) {
            if (i % 3 == 0 && i % 5 == 0) {
                System.out.println("WoTech");
            } else if (i % 3 == 0) {
                System.out.println("Wo");
            } else if (i % 5 == 0) {
                System.out.println("Tech");
            }
        }
    }
}
```
